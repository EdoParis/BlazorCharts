//Render chart and legend using a custom theme
<HistChart Theme="@custom_theme" Model="@hist_model""/>
<LegendBar Theme="@custom_theme" Model="@hist_model""/>

//Or using a built-in static theme for both
<HistChart Theme="Theme.Dark" Model="@hist_model""/>
<LegendBar Theme="Theme.Dark" Model="@hist_model""/>

@{
    //Define you theme.
    private Theme custom_theme = new Theme()
    {
        BackgroundColor = Color.Snow,
        AxisColor = Color.DarkGray,
        TextColor = Color.Black,
        FontFamily = null //Font will be inherited
    };

    //Fill model with data.
    private Histogram hist_model = new Histogram("Asse-X", "Asse-Y", KnownColor.LightSteelBlue, KnownColor.MediumPurple);

    for (int i = 0; i < 10; i++)
    {
        hist_model.Add(new Bin()
        {
            Min = i,
            Max = i + 1,
            Value = 10 + 3 * i
        });
    }
}
