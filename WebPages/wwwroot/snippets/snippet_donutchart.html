<DonutChart Model="@piechart_model" 
        OnClick="OnClickHandler"/>

@{
    private Piegram piechart_model = new Piegram();

    piechart_model.Add(new Slice("Slice1", 10, KnownColor.Red);
    piechart_model.Add(new Slice("Slice2", 20, KnownColor.Orange);
    piechart_model.Add(new Slice("Slice3", 30, KnownColor.Green);

    private void OnClickHandler(Slice slice)
    {
        Console.WriteLine($"clicked slice with value: {slice.Value}");
    }
}
