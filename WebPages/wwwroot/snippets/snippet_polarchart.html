<PolarChart Model="@polarchart_model" 
        OnClick="OnClickHandler"/>

@{
    private Polargram polarchart_model = new Polargram("PolarChart-1");

    polarchart_model.Add(new Slice($"Slice-{i}", 10, KnownColor.Red);
    polarchart_model.Add(new Slice($"Slice-{i}", 20, KnownColor.Orange);
    polarchart_model.Add(new Slice($"Slice-{i}", 30, KnownColor.Green);

    private void OnClickHandler(Slice slice)
    {
        Console.WriteLine($"clicked slice with value: {slice.Value}");
    }
}
