<HistChart Model="@hist_model" OnClick="OnClickHandler"/>

@{
    private Histogram hist_model = new Histogram("Asse-X", "Asse-Y", KnownColor.LightSteelBlue);

    for (int i = 0; i < 10; i++)
    {
        hist_model.Add(new Bin()
        {
            Min = i,
            Max = i + 1,
            Value = 10 + 3 * i
        });
    }

    private void OnClickHandler(Bin bin)
    {
    Console.WriteLine($"clicked bin with value: {bin.Value}");
    }
}
