<LineChart Model="@linechart_model"/>

@{
    private Linegram linechart_model = new Linegram("Asse-X", "Asse-Y");

    List<PointF> points1 = new();
    List<PointF> points2 = new();

    for (int i = 0; i < 10; i++)
    {
        points1.Add(new PointF(i, i));
        points2.Add(new PointF(i, 2 * i));
    }

    linechart_model.Add(new Line("Line1", KnownColor.Red, points1));
    linechart_model.Add(new Line("Line2", KnownColor.Blue, points2, DrawMode.Drawpoints | DrawMode.Drawline));
}
